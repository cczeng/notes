# 常用属性



```typescript
this.gridOptions = {
    suppressRowClickSelection: true, //禁用整行选中
    // 默认表头渲染组件
    defaultColDef: {
        headerComponentFramework: <{ new(): BaseHeaderComponent }>BaseHeaderComponent,
        headerComponentParams: {}
    },
     suppressMovableColumns: false, // 禁止移动
     suppressMenuHide: false,     // 始终显示列标题上 menu 菜单按钮
     headerHeight: 48, // 表头高度
    
     pagination: true, // 是否启用分页
     paginationPageSize: 300, // 每页数据
}
```



## 单独定义表头

```typescript
columnDefs = [
    {
      headerName: '#',
      width: 60,
      cellRenderer: function (params) {
        return params.node.rowIndex + 1;	// 显示行号
      },
      headerComponentFramework: null,	// 设置了默认渲染表头的话需要取消
    },
    { headerName: 'Make', field: 'make', sortable: true, filter: true, checkboxSelection: true },
    { headerName: 'Model', field: 'model', sortable: true, filter: true },
    { headerName: 'Price', field: 'price', sortable: true, filter: true },
    { headerName: 'Age', field: 'age', sortable: true, filter: true },
    { headerName: 'Time', field: 'time', sortable: true, },
    { headerName: 'Memo', field: 'memo', sortable: true, filter: true },
  ];
```
